jQuery Disable Form On Submit
  • View on GitHub

Default Options


$('form[data-disable-on-submit]').disableOnSubmit();

Changing Duration


$('form.disable-on-submit-custom-duration').disableOnSubmit({
    duration: 1000
});

Changing button template


$('form.disable-on-submit-custom-template').disableOnSubmit({
    buttonTemplate: '<span class="mdi mdi-loading mdi-spin"><\span>',
    submitInputText: 'Please wait to try to submit again.'
});

Override Duration with Data- Attribute


$('form[data-disable-on-submit]').disableOnSubmit();

<form class="row mb-4" data-disable-on-submit="1000">
    ...
</form>
